/* Este es un archivo de prueba para el analizador léxico en lenguaje C */
#include <stdio.h>

int main() {
    int x = 10;
    int y = 20;
    char mensaje[] = "Hola, mundo!";

    int resultado = x * y;

    printf("x: %d\n", x);
    printf("y: %.2f\n", y);
    printf("mensaje: %s\n", mensaje);
    printf("resultado: %d\n", resultado);

    return 0;
}
